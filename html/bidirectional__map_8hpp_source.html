262  * Bidirectional associative container that supports efficient lookup in both directions. Neither items of type
263  * ForwardKey nor InverseKey can be modified.

477  /**
478  * Constructs elements in place. If a pair of values with same ForwardKey or same InverseKey already exists
479  * and the corresponding container requires unique keys, then no insertion happens.
480  * @tparam ARGS
481  * @param args arguments used to construct elements
482  * @return std::pair(iterator to inserted element or already existing element, bool whether insertion happened)
483  * @example if std::multiset is used for forward lookup and the map contains the following pair :(a, b)
484  * then inserting (a, b') is possible whereas (a', b) will not be inserted since the inverse lookup is carried
485  * out by std::unordered_map
486  */
487  template<typename ...ARGS>
488  auto emplace(ARGS &&...args) -> std::pair<iterator, bool> {
489  std::pair<ForwardKey, InverseKey> tmp(std::forward<ARGS>(args)...);
490  if constexpr(!impl::traits::is_multimap_v<ForwardMap>) {
491  auto res = find(tmp.first);
492  if (res != end()) {
493  return {res, false};
494  }
495  }
496 
497  if constexpr(!impl::traits::is_multimap_v<InverseMap>) {
498  auto invRes = inverse().find(tmp.second);
499  if (invRes != inverse().end()) {
500  return {find(invRes->second), false};
501  }
502  }
503 
504  auto it = impl::get_first(map.emplace(std::move(tmp.first), nullptr));
505  auto invIt = impl::get_first(inverseAccess->map.emplace(std::move(tmp.second), &it->first));
506  it->second = &invIt->first;
507  return {iterator(it), true};
508  }

609  /**
610  * Erases the element at position pos
611  * @param pos iterator to the element to remove. if pos == end(), this method does nothing
612  * @return iterator pointing to the next element in the container
613  */
614  iterator erase(iterator pos) {
615  if (pos == end()) {
616  return pos;
617  }
618 
619  if constexpr(impl::traits::is_multimap_v<InverseMap>) {
620  auto [curr, end] = inverse().equal_range(pos->second);
621  while (curr != end && &curr->first != pos.it->second.get()) {
622  ++curr;
623  }
624 
625  assert(curr != end);
626  inverseAccess->map.erase(curr.it);
627 
628  } else {
629  inverseAccess->map.erase(inverseAccess->map.find(pos->second));
630  }
631 
632  return iterator(map.erase(pos.it));
633  }

635  /**
636  * Erases all elements with forward key equivalent to key.
637  * @param key
638  * @return number of erased elements
639  */
640  std::size_t erase(const ForwardKey &key) {
641  auto [curr, last] = equal_range(key);
642  std::size_t numErased = 0;
643  while (curr != last) {
644  curr = erase(curr);
645  ++numErased;
646  }
647 
648  return numErased;
649  }
