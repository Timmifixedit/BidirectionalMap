372  * @brief Bidirectional associative container that supports efficient lookup in both directions.
373  * @details This class manages two unidirectional maps in order to enable bidirectional lookup. Neither items of
374  * type ForwardKey nor InverseKey can be modified. The map types for forward and for inverse lookup can be changed.
375  * The following map types are supported and have been tested:
376  * - std::unordered_map (default for both lookup directions)
377  * - std::map
378  * - std::unordered_multimap

380  * @tparam ForwardKey Type of key used for forward lookup
381  * @tparam InverseKey Type of key used for inverse lookup
382  * @tparam ForwardMapType base map container used for forward lookup. Default is std::unordered_map
383  * @tparam InverseMapType base map container used for inverse lookup. Default is std::unordered_map

385  * arguments. Further arguments have to be deducible or have defaults. Using a custom map type not included in the

640  /**
641  * Constructs elements in place. If a pair of values with same ForwardKey or same InverseKey already exists
642  * and the corresponding container requires unique keys, then no insertion happens.
643  * For example, if std::multiset is used for forward lookup and the map contains the following pair :(a, b)
644  * then inserting (a, b') is possible whereas (a', b) will not be inserted since the inverse lookup is carried
645  * out by std::unordered_map
646  * @tparam ARGS argument types
647  * @param args arguments used to construct elements
648  * @return std::pair(iterator to inserted element or already existing element, bool whether insertion happened)
649  */
650  template<typename ...ARGS>
651  auto emplace(ARGS &&...args) -> std::pair<iterator, bool> {
652  std::pair<ForwardKey, InverseKey> tmp(std::forward<ARGS>(args)...);
653  if constexpr(!impl::traits::is_multimap_v<ForwardMap>) {
654  auto res = find(tmp.first);
655  if (res != end()) {
656  return {res, false};
657  }
658  }
659 
660  if constexpr(!impl::traits::is_multimap_v<InverseMap>) {
661  auto invRes = inverse().find(tmp.second);
662  if (invRes != inverse().end()) {
663  return {find(invRes->second), false};
664  }
665  }
666 
667  auto it = impl::get_first(map.emplace(std::move(tmp.first), nullptr));
668  auto invIt = impl::get_first(inverseAccess->map.emplace(std::move(tmp.second), &it->first));
669  it->second = &invIt->first;
670  return {iterator(it), true};
671  }
